30   def eval_statement(instr): 
31       tokens = instr.split('=') 
32       tokens = [tok.strip() for tok in tokens] 
33    
34       # Parse RHS 
35       if len(tokens) == 2:  # assignment 
36    
37           # Check valid variable name 
38           if is_valid_variable_name(tokens[0]): 
39    
40               # Assigns value in dict 
41               var_dict[tokens[0]] = parse_rhs(tokens[1]) 
42               return var_dict[tokens[0]] 
43           else: 
44               raise Exception('Invalid variable name: `' + tokens[0] + '`' 
45                               ) 
46       elif len(tokens) == 1: 
47    
48           # inline arithmetic 
49           if tokens[0]: 
50               return parse_rhs(tokens[0]) 
51           else: 
52               raise Exception('Empty statement.') 
53       else: 
54           raise Exception('Invalid statement `' + instr + '`')

57   def parse_val(token): 
58       if is_int(token): 
59           return int(token) 
60       elif is_valid_variable_name(token): 
61           if token in var_dict and var_dict[token]: 
62               return var_dict[token] 
63           else: 
64               raise Exception('`' + token + '` not declared or defined.') 
65       else: 
66           raise Exception('Invalid value: `' + token + '`')

69   def parse_rhs(rhs): 
70       rhs = rhs.replace(' ', '') 
71    
72       # Check if valid RHS 
73       test = re.match('^\-*(\d+|[A-Za-z]\w*)(?:([/+\-*])\-*(\d+|[A-Za-z]\w*))*$', rhs) 
74       if not test: 
75           raise Exception('Invalid statement: `' + rhs + '`') 
76    
77       # Split RHS into a list 
78       groups = list(re.findall("\-*(\d+|[A-Za-z]\w*|[/+\-*])", rhs)) 
79    
80       # RHS only 1 item 
81       if len(groups) == 1: 
82           return parse_val(groups[0]) 
83    
84       # Multiply & Divide 
85       i = 1 
86       while i < len(groups) - 1: 
87           if groups[i] == '*' or groups[i] == '/': 
88    
89               if groups[i] == '*': 
90                   groups[i - 1] = parse_val(groups[i - 1]) * parse_val(groups[i + 1]) 
91    
92               if groups[i] == '/': 
93                   groups[i - 1] = parse_val(groups[i - 1]) / parse_val(groups[i + 1]) 
94    
95               groups = groups[:i] + groups[i + 2:] 
96           else: 
97               i += 1 
98    
99       # Add & Subtract 
100      i = 1 
101      while i < len(groups) - 1: 
102          if groups[i] == '+' or groups[i] == '-': 
103   
104              if groups[i] == '+': 
105                  groups[i - 1] = parse_val(groups[i - 1]) + parse_val(groups[i + 1]) 
106   
107              if groups[i] == '-': 
108                  groups[i - 1] = parse_val(groups[i - 1]) - parse_val(groups[i + 1]) 
109   
110              groups = groups[:i] + groups[i + 2:] 
111          else: 
112              i += 1 
113   
114      return groups[0]
